integer function int(integer a, integer b)
{
type integer c;
c = 3*(a-b)+3/6;
return c;
}

string function pw(string e)
{
type string d;
type integer o;
d = e;
write(d);
return d;
}

integer function qwerty()
{
type integer c;
c = 1;
return c; 
}

type integer r;
type integer r;
